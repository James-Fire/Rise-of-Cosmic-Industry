decision_build_autofab = {
	owned_planets_only = yes
	icon = decision_autofab
	enactment_time = @autofab_construction_time
	resources = {
		category = planet_districts
		cost = {
			energy = @autofab_cost
			minerals = @autofab_cost
			alloys = @autofaballoycost 
		}
	}
	
	potential = {
		exists = owner
		owner = {
			can_use_autofab = yes
		}
		NOT = { has_deposit = autofab }
	}

	effect = {
		add_deposit = autofab
	}
	
	ai_weight = {
		weight = 1000
	}
}

decision_dismantle_autofab = {
	owned_planets_only = yes
	icon = decision_autofab
	enactment_time = @factory_retool_time
	resources = {
		category = planet_districts
		cost = {
			energy = 200
		}
	}
	
	potential = {
		owner = {
			is_ai = no
		}
		has_deposit = autofab
	}

	effect = {
		remove_deposit = autofab
	}
	
	ai_weight = {
		weight = 0
	}
}
